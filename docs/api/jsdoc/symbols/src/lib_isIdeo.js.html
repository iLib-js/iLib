 21 
 22 var ilib = require("../index.js");
 23 var CType = require("./CType.js");
 24 var IString = require("./IString.js");
 25 
 26 /**
 27  * Return whether or not the first character is an ideographic character.<p>
 28  *
 29  * @static
 30  * @param {string|IString|number} ch character or code point to examine
 31  * @return {boolean} true if the first character is an ideographic character.
 32  */
 33 var isIdeo = function (ch) {
 34     var num;
 35     switch (typeof(ch)) {
 36         case 'number':
 37             num = ch;
 38             break;
 39         case 'string':
 40             num = IString.toCodePoint(ch, 0);
 41             break;
 42         case 'undefined':
 43             return false;
 44         default:
 45             num = ch._toCodePoint(0);
 46             break;
 47     }
 48 
 49     return CType._inRange(num, 'cjk', ilib.data.ctype) ||
 50         CType._inRange(num, 'cjkradicals', ilib.data.ctype) ||
 51         CType._inRange(num, 'enclosedcjk', ilib.data.ctype) ||
 52         CType._inRange(num, 'cjkpunct', ilib.data.ctype) ||
 53         CType._inRange(num, 'cjkcompatibility', ilib.data.ctype);
 54 };
 55 
 56 /**
 57  * @protected
 58  * @param {boolean} sync
 59  * @param {Object|undefined} loadParams
 60  * @param {function(*)|undefined} onLoad
 61  */
 62 isIdeo._init = function (sync, loadParams, onLoad) {
 63     CType._init(sync, loadParams, onLoad);
 64 };
 65 
 66 module.exports = isIdeo;
