 21 var path = {
 22     dirname: function(pathname) {
 23         var i = pathname.lastIndexOf("/");
 24         return i !== -1 ? pathname.substring(0,i) : pathname;
 25     },
 26 
 27     join: function() {
 28         var arr = [];
 29         for (var i = 0; i < arguments.length; i++) {
 30             arr.push(arguments[i]);
 31         }
 32         return arr.join("/");
 33     },
 34 
 35     normalize: function(pathname) {
 36         if (pathname) {
 37             var previousLen;
 38             do {
 39                 previousLen = pathname.length;
 40                 pathname = pathname.replace(/\/\.\//g, "/").replace(/\/[^/]+[^\.]\/\.\./g, "").replace(/^\.\//, "").replace(/\/\//g, "/");
 41             } while (pathname.length < previousLen);
 42         }
 43         return pathname;
 44     }
 45 };

 48 var requireClass = function() {
 49     this.cache = {};
 50     this.loading = {};
 51     this.updateRequire = /\brequire\s*\(\s*"([^/][^"+]*)"/g;
 52 
 53     var pos;
 54     var scripts = document.getElementsByTagName("script");
 55 
 56     this.protocol = "file://";
 57     this.root = ".";
 58 
 59     for (var i = 0; i < scripts.length; i++) {
 60         var source = scripts[i].src;
 61         if (source && (pos = source.search(/ilib-web\.js$/)) !== -1) {
 62             var colon = source.indexOf('://');
 63             this.protocol = source.substring(0,colon+3);
 64             this.root = source.substring(colon+3, pos-1);
 65             break;
 66         }
 67     }
 68 };

 70 requireClass.prototype.loadFile = function(pathname) {
 71     // special case for IE because it has a @#$%ed up XMLHttpRequest implementation
 72     var req = (navigator.userAgent.indexOf(" .NET") > -1) ? 
 73         new ActiveXObject("MSXML2.XMLHTTP") :
 74         new XMLHttpRequest();
 75 
 76     req.open("GET", this.protocol + pathname, false);
 77     req.send();
 78 
 79     return req.responseText;
 80 };

 82 requireClass.prototype.require = function(pathname) {
 83     
 84     //console.log("this.root is " + this.root + " and pathname before was " + pathname);
 85     
 86     if (pathname.charAt(0) !== '/') {
 87         // make the current path be relative to the parent's path, otherwise use the root
 88         pathname = path.join((module && module.filename) ? path.dirname(module.filename) : this.root, pathname);
 89     }
 90 
 91     pathname = path.normalize(pathname);
 92 
 93     if (this.cache[pathname]) {
 94         return this.cache[pathname];
 95     }
 96 
 97     // don't try to load things that are currently in the process of loading
 98     if (this.loading[pathname]) {
 99         //console.log("require: already loading...");
100         return {};
101     }
102 
103     console.log("loading module " + pathname);
104 
105     this.loading[pathname] = true;
106 
107     var text = this.loadFile(pathname);
108     var dirname = path.dirname(pathname);
109     var match, replacement;
110     
111     if (text) {
112         var tmp = module.filename;
113         module.filename = pathname;
114         module.exports = null;
115         module.require = requireClass.prototype.require.bind(r);
116         
117         while ((match = this.updateRequire.exec(text)) !== null) {
118             replacement = path.normalize(path.join(dirname, match[1]));
119             text = text.replace(new RegExp('"' + match[1] + '"', "g"), '"' + replacement + '"');
120             this.updateRequire.lastIndex = match.index + replacement.length + 2;
121         }
122         // for debugging in chrome with nice file names instead of random ids like "VM (2342)"
123         text = text + "\n//# sourceURL=" + pathname + "\n";
124 
125         // console.log("text is " + text);
126         try {
127             eval(text);
128 
129             this.cache[pathname] = module.exports;
130         } finally {
131             this.loading[pathname] = undefined;
132             module.filename = tmp;
133         }
134 
135         return module.exports;
136     }
137 
138     return undefined;
139 };
