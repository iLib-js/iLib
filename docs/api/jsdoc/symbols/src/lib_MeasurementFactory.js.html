 56 /**
 57  * Create a measurement subclass instance based on a particular measure
 58  * required. The measurement is immutable once
 59  * it is created, but it can be converted to other measurements later.<p>
 60  *
 61  * The options may contain any of the following properties:
 62  *
 63  * <ul>
 64  * <li><i>amount</i> - either a numeric amount for this measurement given
 65  * as a number of the specified units, or another Measurement instance
 66  * to convert to the requested units. If converting to new units, the type
 67  * of measure between the other instance's units and the current units
 68  * must be the same. That is, you can only convert one unit of mass to
 69  * another. You cannot convert a unit of mass into a unit of length.
 70  *
 71  * <li><i>unit</i> - units of this measurement. Use the
 72  * static call {@link MeasurementFactory.getAvailableUnits}
 73  * to find out what units this version of ilib supports. If the given unit
 74  * is not a base unit, the amount will be normalized to the number of base units
 75  * and stored as that number of base units.
 76  * For example, if an instance is constructed with 1 kg, this will be converted
 77  * automatically into 1000 g, as grams are the base unit and kg is merely a
 78  * commonly used scale of grams.
 79  * </ul>
 80  *
 81  * Here are some examples of converting a length into new units.
 82  * The first method is via this factory function by passing the old measurement
 83  * in as the "amount" property.<p>
 84  *
 85  * <pre>
 86  * var measurement1 = MeasurementFactory({
 87  *   amount: 5,
 88  *   units: "kilometers"
 89  * });
 90  * var measurement2 = MeasurementFactory({
 91  *   amount: measurement1,
 92  *   units: "miles"
 93  * });
 94  * </pre>
 95  *
 96  * The value in measurement2 will end up being about 3.125 miles.<p>
 97  *
 98  * The second method uses the convert method.<p>
 99  *
100  * <pre>
101  * var measurement1 = MeasurementFactory({
102  *   amount: 5,
103  *   units: "kilometers"
104  * });
105  * var measurement2 = measurement1.convert("miles");
106  * });
107  * </pre>
108  *
109  * The value in measurement2 will again end up being about 3.125 miles.
110  *
111  * @static
112  * @param {Object=} options options that control the construction of this instance
113  */
114 var MeasurementFactory = function(options) {
115     if (!options || typeof(options.unit) === 'undefined') {
116         return undefined;
117     }
118 
119     var measurement, measure = undefined;
120 
121     // first try in the existing case
122     for (var c in Measurement._constructors) {
123         measurement = Measurement._constructors[c];
124         if (Measurement.getUnitId(measurement, options.unit)) {
125             measure = c;
126             break;
127         }
128     }
129 
130     if (!measure) {
131         // if it wasn't found before, try again in lower case -- this may recognize incorrectly because some
132         // units can differ only in their case like "mm" and "Mm"
133         for (var c in Measurement._constructors) {
134             measurement = Measurement._constructors[c];
135             if (typeof(Measurement.getUnitIdCaseInsensitive(measurement, options.unit)) !== 'undefined') {
136                 measure = c;
137                 break;
138             }
139         }
140     }
141 
142     if (!measure || typeof(measure) === 'undefined') {
143         return new UnknownUnit({
144             unit: options.unit,
145             amount: options.amount
146         });
147     } else {
148         return new Measurement._constructors[measure](options);
149     }
150 };

152 /**
153  * Return a list of all possible units that this version of ilib supports.
154  * Typically, the units are given as their full names in English. Unit names
155  * are case-insensitive.
156  *
157  * @static
158  * @return {Array.<string>} an array of strings containing names of measurement
159  * units available
160  */
161 MeasurementFactory.getAvailableUnits = function () {
162     var units = [];
163     for (var c in Measurement._constructors) {
164         var measure = Measurement._constructors[c];
165         units = units.concat(measure.getMeasures());
166     }
167     return units;
168 };
