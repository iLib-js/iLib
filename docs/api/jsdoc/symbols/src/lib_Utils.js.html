 28 /**
 29  * Return the property name inside of ilib.data that corresponds to the given locale data file.
 30  *
 31  * @private
 32  * @param {String} basename the basename of the file
 33  * @param {String} pathname the path from the root to the base file which usually encodes the
 34  * locale of the file
 35  * @param {String=} root the root directory of the file or undefined for the standard locale dir
 36  */
 37 function getPropertyName(basename, pathname, root) {
 38     var bits = [ basename ];
 39     if (root) {
 40         bits = bits.concat(root.split("\/"));
 41     }
 42     if (pathname) {
 43         bits = bits.concat(pathname.split("\/"));
 44     }
 45     return bits.join('_');
 46 }

 48 /**
 49  * Return an array of locales that represent the sublocales of
 50  * the given locale. These sublocales are intended to be used
 51  * to load locale data. Each sublocale might be represented
 52  * separately by files on disk in order to share them with other
 53  * locales that have the same sublocales. The sublocales are
 54  * given in the order that they should be loaded, which is
 55  * least specific to most specific.<p>
 56  *
 57  * For example, the locale "en-US" would have the sublocales
 58  * "root", "en", "und-US", and "en-US".<p>
 59  *
 60  * <h4>Variations</h4>
 61  *
 62  * With only language and region specified, the following
 63  * sequence of sublocales will be generated:<p>
 64  *
 65  * <pre>
 66  * root
 67  * language
 68  * und-region
 69  * language-region
 70  * </pre>
 71  *
 72  * With only language and script specified:<p>
 73  *
 74  * <pre>
 75  * root
 76  * language
 77  * language-script
 78  * </pre>
 79  *
 80  * With only script and region specified:<p>
 81  *
 82  * <pre>
 83  * root
 84  * und-region
 85  * </pre>
 86  *
 87  * With only region and variant specified:<p>
 88  *
 89  * <pre>
 90  * root
 91  * und-region
 92  * region-variant
 93  * </pre>
 94  *
 95  * With only language, script, and region specified:<p>
 96  *
 97  * <pre>
 98  * root
 99  * language
100  * und-region
101  * language-script
102  * language-region
103  * language-script-region
104  * </pre>
105  *
106  * With only language, region, and variant specified:<p>
107  *
108  * <pre>
109  * root
110  * language
111  * und-region
112  * language-region
113  * und-region-variant
114  * language-region-variant
115  * </pre>
116  *
117  * With all parts specified:<p>
118  *
119  * <pre>
120  * root
121  * language
122  * und-region
123  * language-script
124  * language-region
125  * und-region-variant
126  * language-script-region
127  * language-region-variant
128  * language-script-region-variant
129  * </pre>
130  *
131  * @static
132  * @param {Locale|String} locale the locale to find the sublocales for
133  * @return {Array.<string>} An array of locale specifiers that
134  * are the sublocales of the given on
135  */
136 Utils.getSublocales = function(locale) {
137     var ret = ["root"];
138     var loc = typeof(locale) === "string" ? new Locale(locale) : locale;
139     var lang = loc.getLanguage();
140     var region = loc.getRegion();
141     var script = loc.getScript();
142     var variant = loc.getVariant();
143 
144     if (lang) {
145         ret.push(lang);
146     }
147 
148     if (region) {
149         ret.push('und-' + region);
150     }
151 
152     if (lang) {
153         if (script) {
154             ret.push(lang + '-' + script);
155         }
156 
157         if (region) {
158             ret.push(lang + '-' + region);
159         }
160     }
161 
162     if (region && variant) {
163         ret.push("und-" + region + '-' + variant);
164     }
165 
166     if (lang) {
167         if (script && region) {
168             ret.push(lang + '-' + script + '-' + region);
169         }
170 
171         if (region && variant) {
172             ret.push(lang + '-' + region + '-' + variant);
173         }
174 
175         if (script && region && variant) {
176             ret.push(lang + '-' + script + '-' + region + '-' + variant);
177         }
178     }
179     return ret;
180 };

182 /**
183  * Find and merge all the locale data for a particular prefix in the given locale
184  * and return it as a single javascript object. This merges the data in the
185  * correct order:
186  *
187  * <ol>
188  * <li>shared data (usually English)
189  * <li>data for language
190  * <li>data for language + region
191  * <li>data for language + region + script
192  * <li>data for language + region + script + variant
193  * </ol>
194  *
195  * It is okay for any of the above to be missing. This function will just skip the
196  * missing data.
197  *
198  * @static
199  * @param {string} prefix prefix under ilib.data of the data to merge
200  * @param {Locale} locale locale of the data being sought
201  * @param {boolean=} replaceArrays if true, replace the array elements in object1 with those in object2.
202  * If false, concatenate array elements in object1 with items in object2.
203  * @param {boolean=} returnOne if true, only return the most locale-specific data. If false,
204  * merge all the relevant locale data together.
205  * @param {string=} root root path if there is one
206  * @return {Object?} the merged locale data
207  */
208 Utils.mergeLocData = function (prefix, locale, replaceArrays, returnOne, root) {
209     var data = undefined;
210     var loc = locale || new Locale();
211     var mostSpecific;
212 
213     data = {};
214 
215     mostSpecific = data;
216 
217     Utils.getSublocales(loc).forEach(function(l) {
218         var property = getPropertyName(prefix, (l === "root") ? undefined : l.replace(/-/g, "/"), root);
219 
220         if (ilib.data[property]) {
221             if (returnOne) {
222                 mostSpecific = ilib.data[property];
223             } else {
224                 data = JSUtils.merge(data, ilib.data[property], replaceArrays);
225             }
226         }
227     });
228 
229     return returnOne ? mostSpecific : data;
230 };

233 /**
234  * Return an array of relative path names for the
235  * files that represent the data for the given locale.<p>
236  *
237  * Note that to prevent the situation where a directory for
238  * a language exists next to the directory for a region where
239  * the language code and region code differ only by case, the
240  * plain region directories are located under the special
241  * "undefined" language directory which has the ISO code "und".
242  * The reason is that some platforms have case-insensitive
243  * file systems, and you cannot have 2 directories with the
244  * same name which only differ by case. For example, "es" is
245  * the ISO 639 code for the language "Spanish" and "ES" is
246  * the ISO 3166 code for the region "Spain", so both the
247  * directories cannot exist underneath "locale". The region
248  * therefore will be loaded from "und/ES" instead.<p>
249  *
250  * <h4>Variations</h4>
251  *
252  * With only language and region specified, the following
253  * sequence of paths will be generated:<p>
254  *
255  * <pre>
256  * language
257  * und/region
258  * language/region
259  * </pre>
260  *
261  * With only language and script specified:<p>
262  *
263  * <pre>
264  * language
265  * language/script
266  * </pre>
267  *
268  * With only script and region specified:<p>
269  *
270  * <pre>
271  * und/region
272  * </pre>
273  *
274  * With only region and variant specified:<p>
275  *
276  * <pre>
277  * und/region
278  * region/variant
279  * </pre>
280  *
281  * With only language, script, and region specified:<p>
282  *
283  * <pre>
284  * language
285  * und/region
286  * language/script
287  * language/region
288  * language/script/region
289  * </pre>
290  *
291  * With only language, region, and variant specified:<p>
292  *
293  * <pre>
294  * language
295  * und/region
296  * language/region
297  * region/variant
298  * language/region/variant
299  * </pre>
300  *
301  * With all parts specified:<p>
302  *
303  * <pre>
304  * language
305  * und/region
306  * language/script
307  * language/region
308  * region/variant
309  * language/script/region
310  * language/region/variant
311  * language/script/region/variant
312  * </pre>
313  *
314  * @static
315  * @param {Locale} locale load the files for this locale
316  * @param {string?} name the file name of each file to load without
317  * any path
318  * @return {Array.<string>} An array of relative path names
319  * for the files that contain the locale data
320  */
321 Utils.getLocFiles = function(locale, name) {
322     var filename = name || "resources.json";
323     var loc = locale || new Locale();
324 
325     return Utils.getSublocales(loc).map(function(l) {
326         return (l === "root") ? filename : Path.join(l.replace(/-/g, "/"), filename);
327     });
328 };

330 /**
331  * Load data using the new loader object or via the old function callback.
332  * @static
333  * @private
334  */
335 Utils._callLoadData = function (files, sync, params, root, callback) {
336     // console.log("Utils._callLoadData called");
337     if (typeof(ilib._load) === 'function') {
338         // console.log("Utils._callLoadData: calling as a regular function");
339         return ilib._load(files, sync, params, callback);
340     } else if (typeof(ilib._load) === 'object' && typeof(ilib._load.loadFiles) === 'function') {
341         // console.log("Utils._callLoadData: calling as an object");
342         return ilib._load.loadFiles(files, sync, params, callback, root);
343     }
344 
345     // console.log("Utils._callLoadData: not calling. Type is " + typeof(ilib._load) + " and instanceof says " + (ilib._load instanceof Loader));
346     return undefined;
347 };

349 function getPropertyNameFromFile(basename, filepath, root) {
350     var dir = Path.dirname(filepath);
351     return getPropertyName(basename, (dir === "." || dir === "/" || dir === "..") ? undefined : dir, root);
352 }

354 /**
355  * Return true if the locale data corresponding to the given pathname is not already loaded
356  * or assembled.
357  *
358  * @private
359  * @param basename
360  * @param locale
361  * @returns
362  */
363 function dataNotExists(basename, pathname, root) {
364     return !ilib.data[getPropertyNameFromFile(basename, pathname, root)];
365 }

367 /**
368  * Find locale data or load it in. If the data with the given name is preassembled, it will
369  * find the data in ilib.data. If the data is not preassembled but there is a loader function,
370  * this function will call it to load the data. Otherwise, the callback will be called with
371  * undefined as the data. This function will create a cache under the given class object.
372  * If data was successfully loaded, it will be set into the cache so that future access to
373  * the same data for the same locale is much quicker.<p>
374  *
375  * The parameters can specify any of the following properties:<p>
376  *
377  * <ul>
378  * <li><i>name</i> - String. The name of the file being loaded. Default: ResBundle.json
379  * <li><i>object</i> - String. The name of the class attempting to load data. This is used to differentiate parts of the cache.
380  * <li><i>locale</i> - Locale. The locale for which data is loaded. Default is the current locale.
381  * <li><i>nonlocale</i> - boolean. If true, the data being loaded is not locale-specific.
382  * <li><i>type</i> - String. Type of file to load. This can be "json" or "other" type. Default: "json"
383  * <li><i>replace</i> - boolean. When merging json objects, this parameter controls whether to merge arrays
384  * or have arrays replace each other. If true, arrays in child objects replace the arrays in parent
385  * objects. When false, the arrays in child objects are concatenated with the arrays in parent objects.
386  * <li><i>root</i> - String. If provided, look in this root directory first for files, and then fall back
387  * to the standard include paths if they are not found in this root. If not provided, just search the
388  * standard include paths.
389  * <li><i>loadParams</i> - Object. An object with parameters to pass to the loader function
390  * <li><i>sync</i> - boolean. Whether or not to load the data synchronously
391  * <li><i>callback</i> - function(?)=. callback Call back function to call when the data is available.
392  * Data is not returned from this method, so a callback function is mandatory.
393  * </ul>
394  *
395  * @static
396  * @param {Object} params Parameters configuring how to load the files (see above)
397  */
398 Utils.loadData = function(params) {
399     var name = "resources.json",
400         locale = new Locale(ilib.getLocale()),
401         sync = false,
402         type = undefined,
403         loadParams = {},
404         callback = undefined,
405         nonlocale = false,
406         replace = false,
407         root,
408         basename;
409 
410     if (!params || typeof(params.callback) !== 'function') {
411         throw "Utils.loadData called without a callback. It must have a callback to work.";
412     }
413 
414     if (params.name) {
415         name = params.name;
416     }
417     if (params.locale) {
418         locale = (typeof(params.locale) === 'string') ? new Locale(params.locale) : params.locale;
419     }
420     if (params.type) {
421         type = params.type;
422     }
423     if (params.loadParams) {
424         loadParams = params.loadParams;
425     }
426     if (params.sync) {
427         sync = params.sync;
428     }
429     if (params.nonlocale) {
430         nonlocale = !!params.nonlocale;
431     }
432     if (typeof(params.replace) === 'boolean') {
433         replace = params.replace;
434     }
435 
436     root = params.root;
437     callback = params.callback;
438 
439     if (!type) {
440         var dot = name.lastIndexOf(".");
441         type = (dot !== -1) ? name.substring(dot+1) : "text";
442     }
443 
444     if (typeof(ilib.data.cache) === "undefined") {
445         ilib.data.cache = {};
446     }
447     if (typeof(ilib.data.cache.fileSet) === "undefined") {
448         ilib.data.cache.fileSet = new ISet();
449     }
450 
451     var data, returnOne = ((loadParams && loadParams.returnOne) || type !== "json");
452 
453     basename = name.substring(0, name.lastIndexOf(".")).replace(/[\.:\(\)\/\\\+\-]/g, "_");
454 
455     if (ilib._cacheMerged) {
456         if (typeof(ilib.data.cache.merged) === "undefined") {
457             ilib.data.cache.merged = {};
458         }
459         var spec = ((!nonlocale && locale.getSpec().replace(/-/g, '_')) || "root") + "," + basename + "," + String(JSUtils.hashCode(loadParams));
460         if (typeof(ilib.data.cache.merged[spec]) !== 'undefined') {
461             // cache hit!
462             callback(ilib.data.cache.merged[spec]);
463             return;
464         }
465     }
466 
467     if (typeof(ilib._load) !== 'undefined') {
468         // We have a loader, so we can figure out which json files are loaded already and
469         // which are not so that we can load the missing ones.
470         // the data is not preassembled, so attempt to load it dynamically
471         var files = nonlocale ? [ name || "resources.json" ] : Utils.getLocFiles(locale, name);
472 
473         // find the ones we haven't loaded before
474         files = files.filter(ilib.bind(this, function(file) {
475             return !ilib.data.cache.fileSet.has(Path.join(root, file)) && dataNotExists(basename, file, root);
476         }));
477 
478         if (files.length) {
479             Utils._callLoadData(files, sync, loadParams, root, ilib.bind(this, function(arr) {
480                 for (var i = 0; i < files.length; i++) {
481                     if (arr[i]) {
482                         var property = nonlocale ? basename : getPropertyNameFromFile(basename, files[i], root);
483 
484                         if (!ilib.data[property]) {
485                             ilib.data[property] = arr[i];
486                         }
487                     }
488                     ilib.data.cache.fileSet.add(Path.join(root, files[i]));
489                 }
490 
491                 if (!nonlocale) {
492                     data = Utils.mergeLocData(basename, locale, replace, returnOne, root);
493                     if (ilib._cacheMerged) ilib.data.cache.merged[spec] = data;
494                 } else {
495                     data = ilib.data[basename];
496                 }
497 
498                 callback(data);
499             }));
500 
501             return;
502         }
503         // otherwise the code below will return the already-loaded data
504     }
505 
506     // No loader, or data already loaded? Then use whatever data we have already in ilib.data
507     if (!nonlocale) {
508         data = Utils.mergeLocData(basename, locale, replace, returnOne, root);
509         if (ilib._cacheMerged) ilib.data.cache.merged[spec] = data;
510     } else {
511         data = ilib.data[basename];
512     }
513 
514     callback(data);
515 };
