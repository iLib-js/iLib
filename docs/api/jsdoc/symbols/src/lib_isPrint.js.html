 20 var isCntrl = require("./isCntrl.js");
 21 
 22 /**
 23  * Return whether or not the first character is any printable character,
 24  * including space.<p>
 25  *
 26  * @static
 27  * @param {string|IString|number} ch character or code point to examine
 28  * @return {boolean} true if the first character is printable.
 29  */
 30 var isPrint = function (ch) {
 31     return typeof(ch) !== 'undefined' && ch.length > 0 && !isCntrl(ch);
 32 };
 33 
 34 /**
 35  * @protected
 36  * @param {boolean} sync
 37  * @param {Object|undefined} loadParams
 38  * @param {function(*)|undefined} onLoad
 39  */
 40 isPrint._init = function (sync, loadParams, onLoad) {
 41     isCntrl._init(sync, loadParams, onLoad);
 42 };
 43 
 44 module.exports = isPrint;
