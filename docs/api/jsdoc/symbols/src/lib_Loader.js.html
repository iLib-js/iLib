 20 var Path = require("./Path.js");
 21 var ilib = require("./ilib.js");
 22 
 23 /**
 24  * @class
 25  * Superclass of the loader classes that contains shared functionality.
 26  *
 27  * @private
 28  * @constructor
 29  */
 30 var Loader = function() {
 31     // console.log("new Loader instance");
 32 
 33     this.protocol = "file://";
 34     this.includePath = [];
 35 };
 36 
 37 Loader.prototype = new ilib.Loader();
 38 Loader.prototype.parent = ilib.Loader;
 39 Loader.prototype.constructor = Loader;
 40 
 41 Loader.prototype._loadFile = function (pathname, sync, cb) {};
 42 
 43 Loader.prototype._exists = function(dir, file) {
 44     var fullpath = Path.normalize(Path.join(dir, file));
 45     if (this.protocol !== "http://") {
 46         var text = this._loadFile(fullpath, true);
 47         if (text && this.includePath.indexOf(text) === -1) {
 48             this.includePath.push(dir);
 49         }
 50     } else {
 51         // put the dir on the list now assuming it exists, and check for its availability
 52         // later so we can avoid the 404 errors eventually
 53         if (this.includePath.indexOf(dir) === -1) {
 54             this.includePath.push(dir);
 55             this._loadFile(fullpath, false, ilib.bind(this, function(text) {
 56                 if (!text) {
 57                     //console.log("Loader._exists: removing " + dir + " from the include path because it doesn't exist.");
 58                     this.includePath = this.includePath.slice(-1);
 59                 }
 60             }));
 61         }
 62     }
 63 };
 64 
 65 Loader.prototype._loadFileAlongIncludePath = function(includePath, pathname) {
 66     for (var i = 0; i < includePath.length; i++) {
 67         var manifest = this.manifest[includePath[i]];
 68         if (!manifest || Loader.indexOf(manifest, pathname) > -1) {
 69             var filepath = Path.join(includePath[i], pathname);
 70             //console.log("Loader._loadFileAlongIncludePath: attempting sync load " + filepath);
 71             var text = this._loadFile(filepath, true);
 72             if (text) {
 73                 //console.log("Loader._loadFileAlongIncludePath: succeeded");
 74                 return text;
 75             }
 76             //else {
 77                 //console.log("Loader._loadFileAlongIncludePath: failed");
 78             //}
 79         }
 80         //else {
 81             //console.log("Loader._loadFileAlongIncludePath: " + pathname + " not in manifest for " + this.includePath[i]);
 82         //}
 83     }
 84 
 85     //console.log("Loader._loadFileAlongIncludePath: file not found anywhere along the path.");
 86     return undefined;
 87 };

 89 Loader.prototype.loadFiles = function(paths, sync, params, callback, root) {
 90     root = root || (params && params.base);
 91     var includePath = root ? [root].concat(this.includePath) : this.includePath;
 92 
 93     //console.log("Loader loadFiles called");
 94     // make sure we know what we can load
 95     if (!paths) {
 96         // nothing to load
 97         //console.log("nothing to load");
 98         return;
 99     }
100 
101     //console.log("generic loader: attempting to load these files: " + JSON.stringify(paths));
102     if (sync) {
103         var ret = [];
104 
105         // synchronous
106         this._loadManifests(true);
107 
108         for (var i = 0; i < paths.length; i++) {
109             var text = this._loadFileAlongIncludePath(includePath, Path.normalize(paths[i]));
110             ret.push(typeof(text) === "string" ? JSON.parse(text) : text);
111         };
112 
113         // only call the callback at the end of the chain of files
114         if (typeof(callback) === 'function') {
115             callback(ret);
116         }
117 
118         return ret;
119     }
120 
121     // asynchronous
122     this._loadManifests(false, ilib.bind(this, function() {
123         //console.log("Loader.loadFiles: now loading files asynchronously");
124         var results = [];
125         this._loadFilesAsync(includePath, paths, results, callback);
126     }));
127 };
128 
129 Loader.prototype._loadFilesAsyncAlongIncludePath = function (includes, filename, cb) {
130     var text = undefined;
131 
132     if (includes.length > 0) {
133         var root = includes[0];
134         includes = includes.slice(1);
135 
136         var manifest = this.manifest[root];
137         if (!manifest || Loader.indexOf(manifest, filename) > -1) {
138             var filepath = Path.join(root, filename);
139             this._loadFile(filepath, false, ilib.bind(this, function(t) {
140                 //console.log("Loader._loadFilesAsyncAlongIncludePath: loading " + (t ? " success" : " failed"));
141                 if (t) {
142                     cb(t);
143                 } else {
144                     this._loadFilesAsyncAlongIncludePath(includes, filename, cb);
145                 }
146             }));
147         } else {
148             //console.log("Loader._loadFilesAsyncAlongIncludePath: " + filepath + " not in manifest for " + root);
149             this._loadFilesAsyncAlongIncludePath(includes, filename, cb);
150         }
151     } else {
152         // file not found in any of the include paths
153         cb();
154     }
155 };
156 
157 Loader.prototype._loadFilesAsync = function (includePath, paths, results, callback) {
158     if (paths.length > 0) {
159         var filename = paths[0];
160         paths = paths.slice(1);
161 
162         //console.log("Loader._loadFilesAsync: attempting to load " + filename + " along the include path.");
163         this._loadFilesAsyncAlongIncludePath(includePath, filename, ilib.bind(this, function (json) {
164             results.push(typeof(json) === "string" ? JSON.parse(json) : json);
165             this._loadFilesAsync(includePath, paths, results, callback);
166         }));
167     } else {
168         // only call the callback at the end of the chain of files
169         if (typeof(callback) === 'function') {
170             callback(results);
171         }
172     }
173 };
174 
175 Loader.prototype._loadManifestFile = function(i, sync, cb) {
176     //console.log("Loader._loadManifestFile: Checking include path " + i + " " + this.includePath[i]);
177     if (i < this.includePath.length) {
178         var filepath = Path.join(this.includePath[i], "ilibmanifest.json");
179         //console.log("Loader._loadManifestFile: Loading manifest file " + filepath);
180         var text = this._loadFile(filepath, sync, ilib.bind(this, function(text) {
181             if (text) {
182                 //console.log("Loader._loadManifestFile: success!");
183                 this.manifest[this.includePath[i]] = (typeof(text) === "string" ? JSON.parse(text) : text).files;
184             }
185             //else console.log("Loader._loadManifestFile: failed...");
186             this._loadManifestFile(i+1, sync, cb);
187         }));
188     } else {
189         if (typeof(cb) === 'function') {
190             //console.log("Loader._loadManifestFile: now calling callback function");
191             cb();
192         }
193     }
194 };
195 
196 Loader.prototype._loadManifests = function(sync, cb) {
197     //console.log("Loader._loadManifests: called " + (sync ? "synchronously" : "asychronously."));
198     if (!this.manifest) {
199         //console.log("Loader._loadManifests: attempting to find manifests");
200         this.manifest = {};
201         if (typeof(sync) !== 'boolean') {
202             sync = true;
203         }
204 
205         this._loadManifestFile(0, sync, cb);
206     } else {
207         //console.log("Loader._loadManifests: already loaded");
208         if (typeof(cb) === 'function') {
209             //console.log("Loader._loadManifests: now calling callback function");
210             cb();
211         }
212     }
213 };
214 
215 Loader.prototype.listAvailableFiles = function(sync, cb) {
216     //console.log("generic loader: list available files called");
217     this._loadManifests(sync, ilib.bind(this, function () {
218         if (typeof(cb) === 'function') {
219             //console.log("generic loader: now calling caller's callback function");
220             cb(this.manifest);
221         }
222     }));
223     return this.manifest;
224 };
225 
226 Loader.indexOf = function(array, obj) {
227     if (!array || !obj) {
228         return -1;
229     }
230     if (typeof(array.indexOf) === 'function') {
231         return array.indexOf(obj);
232     } else {
233         for (var i = 0; i < array.length; i++) {
234             if (array[i] === obj) {
235                 return i;
236             }
237         }
238         return -1;
239     }
240 };
241 
242 Loader.prototype.checkAvailability = function(file) {
243     for (var dir in this.manifest) {
244         if (Loader.indexOf(this.manifest[dir], file) !== -1) {
245             return true;
246         }
247     }
248 
249     return false;
250 };
251 
252 Loader.prototype.isAvailable = function(file, sync, cb) {
253     //console.log("Loader.isAvailable: called");
254     if (typeof(sync) !== 'boolean') {
255         sync = true;
256     }
257     if (sync) {
258         this._loadManifests(sync);
259         return this.checkAvailability(file);
260     }
261 
262     this._loadManifests(false, ilib.bind(this, function () {
263         // console.log("generic loader: isAvailable " + path + "? ");
264         if (typeof(cb) === 'function') {
265             cb(this.checkAvailability(file));
266         }
267     }));
268 };
269 
270 module.exports = Loader;
